import pytest

def test_valid_query():
    result = query_engine.execute(query="品牌声誉", platforms=["all"])
    assert len(result) >= 5
    assert result[0]["source"] is not None
    assert result[0]["url"] is not None

def test_empty_query():
    with pytest.raises(SYSTEMError) as exc:
        query_engine.execute(query="", platforms=["all"])
    assert exc.value.code == "QUERY_001"

def test_keyword_optimization():
    keywords = query_engine.optimize_keywords("品牌声誉")
    assert "品牌" in keywords
    assert "声誉" in keywords

    raw_data = [{"url": "a.com"}, {"url": "a.com"}, {"url": "b.com"}]
    deduped = query_engine.deduplicate(raw_data)
    assert len(deduped) == 2

## Integration Tests

### Full Pipeline Integration Test

```python
# test_full_pipeline.py
import pytest

def test_sentiment_analysis_pipeline():
    # Step 1: QueryEngine
    raw_data = query_engine.execute(
        query="品牌声誉",
        platforms=["weibo", "xiaohongshu"],
        date_range={"start": "2026-04-22", "end": "2026-04-29"}
    )
    assert len(raw_data) >= 5
    
    # Step 2: MediaEngine
    annotated = media_engine.analyze_batch(raw_data)
    assert all("sentiment" in item for item in annotated)
    
    # Step 3: InsightEngine
    insights = insight_engine.process(annotated)
    assert "correlation" in insights
    assert "trends" in insights
    
    # Step 4: ReportEngine
    report = report_engine.generate(insights, format="html")
    assert report is not None
    assert "Generated by AI" in report
    
    # Step 5: ForumEngine
    forum = forum_engine.init_forum(report_id=report.trace_id)
    discussion = forum_engine.run_discussion(forum)
    assert "divergence_summary" in discussion
```
